apiVersion: apps/v1
kind: Deployment
metadata:
  name: auth-users-deployment
spec:
  replicas: 1
  selector:
    matchLabels:
      app: auth-users-pod
  template:
    metadata:
      labels:
        app: auth-users-pod
    spec:
      containers:
        - name: users-container
          image: joydipdocker/users-api-image:1.0.0

          env:

              valueFrom:
                configMapKeyRef:
                  name: key-value-pairs
                  key: authportno

              value: localhost

            limits:
              memory: "256Mi"
              cpu: "500m"
            requests:
              memory: "128Mi"
              cpu: "250m"

          image: joydipdocker/auth-api-image:1.0.0
